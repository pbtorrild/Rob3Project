void MotorSwitch(int &MotorNR, int DD,bool &timeCehk,unsigned long &Time){
    if (DD>800 && timeCehk==false){
        MotorNR = MotorNR + 1;
        timeCehk=true;
        Time = millis();
    }
     if (DD < 450 && timeCehk==false){
        MotorNR = MotorNR - 1;
        timeCehk=true;
        Time = millis();
    }
    if (millis() - Time >= 300){
      timeCehk=false;
    }
    if (MotorNR > 4){
        MotorNR = 1;
    }
     if (MotorNR < 1){
        MotorNR = 4;
    } 
    else{}
 }
